Question 4: What are the counts of the major genres like Action, Comedy, Horror, Documentary, Drama? Which of these tends to have the highest revenue, in terms of the average of the revenue of all the movies in the categories?¶

In [26]:
def average_revenue(categories):
    result = {}
    df = pd.read_csv('tmdb-movies.csv').dropna()
    
    for category in categories:
        df = df[df['genres'].str.contains(category)]
        total = float(df['revenue'].sum())
        count = len(df.index)
        result[category] = total / count
    return result

categories = ['Action', 'Comedy', 'Horror', 'Science Fiction']
average = average_revenue(categories)
